CREATE TABLE faculty(fid INT(5) NOT NULL PRIMARY KEY AUTO_INCREMENT,name VARCHAR(50) NOT NULL,email VARCHAR(50) NOT NULL,ph_no VARCHAR(13) NOT NULL, address VARCHAR(200) );
CREATE TABLE emp(emp_id INT(5) PRIMARY KEY AUTO_INCREMENT,fid INT(5) NOT NULL,salary DOUBLE,join_date DATE,CONSTRAINT `emp_fk` FOREIGN KEY (fid) REFERENCES faculty(fid));
CREATE TABLE courses(fid INT(5) NOT NULL, cid INT(5) PRIMARY KEY AUTO_INCREMENT,cname VARCHAR(100) NOT NULL,technology VARCHAR(200),CONSTRAINT `courses_fk` FOREIGN KEY (fid) REFERENCES faculty(fid) );
